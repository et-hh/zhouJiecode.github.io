    /**
 * @method 动态加载js
 * @param src Array|String
 * @param cb 加载完成回调
 * */
Vue.prototype.loadJs = function (src, cb) {
    let load = (src, cb) => {
        let s = document.createElement('script')
        s.src = src
        s.onload = () => {
            this.isFunction(cb) && cb()
        }
        document.body.appendChild(s)
    }
    let loadedCount = 0
    
    if(this.isArray(src)) {
        src.forEach((s) => {
            load(s, () => {
                if(++loadedCount >= src.length)
                    this.isFunction(cb) && cb()
            })
        })
    } else if(typeof src === 'string') {
        load(src, cb)
    } else {
        console.error('加载的链接不是字符串：' + src)
    }
}

/**
 * 从数组中匹配某个特定属性为特定值的item
 * @return 索引
 * */
Vue.prototype.getIndexByParam = function (v, arr, param) {
    if (!this.isArray(arr) || !v) return -1
    if (!param) return arr.indexOf(v)

    for (let i = 0; i < arr.length; i++) {
        let item = arr[i]
        if (item[param] === v) return i
    }
    return -1
}

findElmByParam: function (v, arr, param) {
      if (!$.isArray(arr) || !v) return null;

      var index = $.inArray(v, arr);
      if (!param) return index === -1 ? null : arr[index];

      for (var i = 0; i < arr.length; i++) {
           var item = arr[i];
           if (item[param] === v) return item;
      }
      return null;
}

/**
 * 函数节流，只执行最后一次
 */
Vue.prototype.debounce = function(action, idle) {
    var last
    return function () {
        var ctx = this, args = arguments
        clearTimeout(last)
        last = setTimeout(() => {
            action.apply(ctx, args)
        }, idle)
    }
}

/**
* 限制执行频率
*/
throttle: function(action, delay){
      var last = 0;
      return function(){
           var curr = +new Date();
           if (curr - last > delay){
                action.apply(this, arguments);
                last = curr ;
           }
     };
}

placeholder: function (doms) {
      if (!-[1,]) {
           //延时用于等页面渲染完毕
           setTimeout(function () {
                $(doms).each(function (k, dom) {
                      var $input = $(dom),
                            $placeHolder = $input.parent().prepend('<span class="holder">' + $input.attr("placeholder") + '</span>').find('.holder');

                        $input.bind("propertychange", function () {
                            $placeHolder.css({"display": $(this).val() === "" && $(this).css('display') !== 'none' ? "block" : "none"});
                        });

                        $placeHolder.bind("click", function () {
                            $input.focus();
                        });

                        $placeHolder.css({"display": $input.val() === "" ? 'block' : "none"});
                    });
           }, 500);
     }
}

    # ie8下锚点可能影响document.title, 使用js还原title

    //ie8下锚点可能影响document.title, 使用js还原title
if (!-[1,]) {
    document.attachEvent('onpropertychange', function (evt) {
        evt = evt || window.event;
        if (evt.propertyName === 'title' && document.title !== constant.DOCUMENT_TITLE) {
            setTimeout(function () {
                document.title = constant.DOCUMENT_TITLE;
            }, 1);
        }
    });
}

    # 按backspace键时，阻止浏览器url回退（返回上一页面）的默认事件

    //在disable或者readonly输入框按backspace键时，阻止浏览器url回退（返回上一页面）的默认事件
$(document).on('keydown', function (event) {
    if (event.keyCode == 8) {
        var target = $(event.target);
        if(!target.is('textarea') && !target.is('input') && !target.is('[contenteditable]'))
            event.preventDefault();
        else if(target.is('[readonly]') || target.is('input[disabled]') || target.is('textarea[disabled]'))
            event.preventDefault();
    }
});

    # 使用a标签解析url

    function parseUrl(url) {
    let a = document.createElement('a');
    a.href = url;
    return {
        host: a.hostname,
        port: a.port,
        query: a.search,
        hash: a.hash.replace('#', ''),
        params: (() => {
            let searchArr = a.search.replace(/^\?/, '').split('&');
            let params = {};
            searchArr.forEach(item => {
                let [key, value] = item.split('=');
                params[key] = params[key] ? [params[key], value].flat() : value;
            });
            return params;
        })()
    }
}
