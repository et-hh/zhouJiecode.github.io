    listen      80;
    listen      443 ssl; # managed by Certbot
    server_name bbs.yiwise.com;
    charset utf-8;

    location / {
       root /home/admin/yiwise-official-BBS/dist; //代理本地代码
       #        proxy_pass  http://127.0.0.1:8080; // 反向代理本机服务
        autoindex on;
        autoindex_exact_size off;
        autoindex_localtime on;
        add_header Cache-Control no-store;
    }

    location /article {
        proxy_pass http://127.0.0.1:13002;
        proxy_set_header Host $host;
    }

    location /apiBlog {
        proxy_pass http://127.0.0.1:13002;
        proxy_set_header Host $host;
    }

    location /api {
        proxy_pass http://backend.yiwise.com:8100;
        proxy_set_header Host $host;
    }

    ssl_certificate /etc/letsencrypt/live/yiwise.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/yiwise.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}
